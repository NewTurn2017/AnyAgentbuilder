@dataclass
class PublicContext:
    public_context: dict[str, Any] = field(default_factory=dict)

    def as_dict(self) -> dict[str, Any]:
        return {key: self.public_context.get(key) for key in PUBLIC_FIELD_NAMES}


@dataclass
class InternalContext:
    internal_context: dict[str, Any] = field(default_factory=dict)

    def as_dict(self) -> dict[str, Any]:

def build_public_context(seed: dict[str, Any]) -> dict[str, Any]:
    public_context = PublicContext(seed)
    return public_context.as_dict()

    return internal_context.as_dict()
