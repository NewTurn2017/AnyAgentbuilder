   101	function parseArgs(argv) {
   102	  const parsed = {
   103	    checkSkill: null,
   104	    validateExample: null,
   105	    requireDomains: [],
   106	    checkGenerated: null,
   107	  };
   108	
   109	  for (let index = 0; index < argv.length; index += 1) {
   110	    const arg = argv[index];
   111	    if (arg === "--check-skill") {
   112	      parsed.checkSkill = readValue(argv, ++index, arg);
   113	    } else if (arg === "--validate-example") {
   114	      parsed.validateExample = readValue(argv, ++index, arg);
   115	    } else if (arg === "--require-domain") {
   116	      parsed.requireDomains.push(readValue(argv, ++index, arg));
   117	    } else if (arg === "--check-generated") {
   118	      parsed.checkGenerated = readValue(argv, ++index, arg);
   119	    } else {
   120	      fail(`unknown flag or argument: ${arg}`);
   121	    }
   122	  }
   123	
   124	  const modes = [parsed.checkSkill, parsed.validateExample, parsed.checkGenerated].filter(Boolean);
   125	  if (modes.length !== 1) {
   126	    fail("choose exactly one mode: --check-skill <dir>, --validate-example <file>, or --check-generated <dir>");
   127	  }
   128	  if (parsed.requireDomains.length > 0 && !parsed.validateExample) {
   129	    fail("--require-domain can only be used with --validate-example <file>");
   130	  }
   131	  if (parsed.validateExample && parsed.requireDomains.length === 0) {
   132	    fail("--validate-example requires at least one --require-domain <domain>");
   133	  }
   134	
   135	  return parsed;
   136	}

   142	  }
   143	  return value;

   204	  const links = markdownLinks(skillText);

   229	    fail(`SKILL.md has unsupported local links; only REFERENCE.md and EXAMPLES.md are allowed: ${bad.join(", ")}`);

   270	function checkSkill(dir) {
   271	  if (!existsSync(dir) || !statSync(dir).isDirectory()) {
   272	    fail(`missing skill directory: ${dir}`);
   273	  }
   274	
   275	  const skillPath = join(dir, "SKILL.md");
   276	  const text = readRequiredFile(skillPath, "SKILL.md");
   277	  const lines = text.split(/\r?\n/);
   278	  if (lines.length > 100) {
   279	    fail(`SKILL.md must be <=100 lines; found ${lines.length}`);
   280	  }
   281	
   282	  const frontmatter = parseFrontmatter(text);
   283	  if (!frontmatter.name) {
   284	    fail("SKILL.md frontmatter missing name");
   285	  }
   286	  if (!frontmatter.description) {
   287	    fail("SKILL.md frontmatter missing description");
   288	  }
   289	  if (frontmatter.description.length > 1024) {
   290	    fail(`SKILL.md description must be <=1024 characters; found ${frontmatter.description.length}`);
   291	  }
   292	  if (!frontmatter.description.includes("Use when")) {
   293	    fail('SKILL.md description must contain "Use when"');
   294	  }
   295	  assertNoTimeSensitiveWording(text, "SKILL.md");
   296	  assertRequiredSkillHeadings(text);
   297	  assertSkillLinks(text, dir);
   298	
   299	  for (const ref of ["REFERENCE.md", "EXAMPLES.md"]) {
   300	    if (!text.includes(ref)) {
   301	      fail(`SKILL.md must reference ${ref}`);
   302	    }
   303	    const refPath = join(dir, ref);
   304	    if (!existsSync(refPath)) {
   305	      fail(`SKILL.md references missing ${ref}: ${refPath}`);
   306	    }
   307	  }
   308	
   309	  const referenceText = readRequiredFile(join(dir, "REFERENCE.md"), "REFERENCE.md");
   310	  const examplesText = readRequiredFile(join(dir, "EXAMPLES.md"), "EXAMPLES.md");
   311	  assertNoTimeSensitiveWording(referenceText, "REFERENCE.md");
   312	  assertNoTimeSensitiveWording(examplesText, "EXAMPLES.md");
   313	  assertNoNestedReferenceChains(referenceText, "REFERENCE.md");
   314	  assertNoNestedReferenceChains(examplesText, "EXAMPLES.md");
   315	  assertReferenceContextSeparation(referenceText, examplesText);
   316	
   317	  ok(`skill structure valid: ${skillPath}`);
   318	}
   371	function assertDomainSchema(block, domain) {
   372	  const keys = extractTopLevelKeys(block);
   373	  const missingTerms = SCHEMA_TERMS.filter((term) => !keys.has(term));
   374	  if (missingTerms.length > 0) {
   375	    fail(`domain ${domain} missing required schema terms: ${missingTerms.join(", ")}`);
   376	  }
   377	
   378	  const publicFields = extractYamlList(block, "public_context");
   379	  const internalFields = extractYamlList(block, "internal_context");
   380	  if (publicFields.length === 0 || internalFields.length === 0) {
   381	    fail(`domain ${domain} must define non-empty public_context and internal_context lists`);
   382	  }
   383	
   384	  const publicSet = new Set(publicFields);
   385	  const overlap = internalFields.filter((field) => publicSet.has(field));
   386	  if (overlap.length > 0) {
   387	    fail(`domain ${domain} leaks internal fields into public_context: ${overlap.join(", ")}`);
   388	  }
   389	
   390	  if (domain === "library") {
   391	    const missingPublic = LIBRARY_PUBLIC_FIELDS.filter((field) => !publicSet.has(field));
   392	    const internalSet = new Set(internalFields);
   393	    const missingInternal = LIBRARY_INTERNAL_FIELDS.filter((field) => !internalSet.has(field));
   394	    if (missingPublic.length > 0) {
   395	      fail(`library example missing required public context fields: ${missingPublic.join(", ")}`);
   396	    }
   397	    if (missingInternal.length > 0) {
   398	      fail(`library example missing required internal context fields: ${missingInternal.join(", ")}`);
   399	    }
   400	  }
   401	}
